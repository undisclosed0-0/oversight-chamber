  return 'CARD-' + Math.random().toString(36).substr(2, 8).toUpperCase();

  return 'ACCESS-' + Math.floor(100000 + Math.random() * 900000);

document.addEventListener("DOMContentLoaded", function () {
  document.addEventListener("keydown", function (e) {
    // Unlock Admin Form: Ctrl + Shift + A
    if (e.ctrlKey && e.shiftKey && e.key.toLowerCase() === "a") {
      const adminForm = document.getElementById("adminForm");
      if (adminForm) {
        adminForm.style.display = "block";
        alert("Admin access form unlocked.");
      }
    }

    // Unlock Super Form: Ctrl + Alt + M
    if (e.ctrlKey && e.altKey && e.key.toLowerCase() === "q") {
      const superForm = document.getElementById("superForm");
      if (superForm) {
        superForm.style.display = "block";
        alert("Super-level access form unlocked.");
      }
    }
  });
});

  <script>
let adminSequence = [];
let superSequence = [];
let adminTimer, superTimer;
let colorSequence = [];

const shimmer = document.getElementById('unlockShimmer');
const sound = document.getElementById('unlockSound');

// Ritual shimmer + sound
function triggerRitual(formId) {
  document.getElementById(formId).style.display = 'block';
  shimmer.style.opacity = '1';
  sound.play();
  setTimeout(() => shimmer.style.opacity = '0', 3000);
}

// Reset sequences
function resetAdminSequence() {
  adminSequence = [];
  clearTimeout(adminTimer);
}
function resetSuperSequence() {
  superSequence = [];
  clearTimeout(superTimer);
}

// Admin unlock: Top Left ‚Üí Top Right
function checkAdminUnlock() {
  const correct = ['tapTopLeft', 'tapTopRight'];
  if (adminSequence.join(',') === correct.join(',')) {
    triggerRitual('adminForm');
    resetAdminSequence();
  } else {
    adminTimer = setTimeout(resetAdminSequence, 10000);
  }
}



// Tap listeners
document.getElementById('tapTopLeft').addEventListener('click', () => {
  adminSequence.push('tapTopLeft');
  checkAdminUnlock();

  superSequence.push('tapTopLeft');
  checkSuperUnlock();
});
document.getElementById('tapTopRight').addEventListener('click', () => {
  adminSequence.push('tapTopRight');
  checkAdminUnlock();

  superSequence.push('tapTopRight');
  checkSuperUnlock();
});
document.getElementById('crestCore').addEventListener('click', () => {
  superSequence.push('crestCore');
  checkSuperUnlock();
});



document.querySelectorAll('.colorBtn').forEach(btn => {
  btn.addEventListener('click', () => {
    colorSequence.push(btn.dataset.color);
    if (colorSequence.length === 3) {
      checkColorUnlock();
    }
  });
});

function checkColorUnlock() {
  const correct = ['gold', 'violet', 'silver'];
  if (colorSequence.join(',') === correct.join(',')) {
    triggerRitual('superForm');
  } else {
    alert('Incorrect combination. Try again.');
    colorSequence = [];
  }
}
document.getElementById('verifyTierBtn').addEventListener('click', () => {
  const code = document.getElementById("adminCode").value.trim();
  if (code !== "ADMIN-ACCESS") {
    alert("Invalid admin clearance code.");
    return;
  }

  document.getElementById('colorPanel').style.display = 'block';
  colorSequence = [];
});
